663

664

665

822 %%-----------------------------------------------------------------------
823 %% recvd_packet
824 %% Deals with incoming packets in the active state
825 %% Will return one of:
826 %% {ok, NewS} - Don't reply to client yet as this is part of a search
827 %% result and we haven't got all the answers yet.
828 %% {reply, Result, From, NewS} - Reply with result to client From
829 %% {error, Reason}
830 %% {'EXIT', Reason} - Broke
831 %%-----------------------------------------------------------------------
832 recvd_packet(Pkt, S) ->
833 
:-(
 case 'ELDAPv3':decode('LDAPMessage', Pkt) of
834 {ok, Msg} ->
835 
:-(
 Op = Msg#'LDAPMessage'.protocolOp,
836 
:-(
 ?DEBUG("~p", [Op]),
837 
:-(
 Dict = S#eldap.dict,
838 
:-(
 Id = Msg#'LDAPMessage'.messageID,
839 
:-(
 {Timer, From, Name, Result_so_far} = get_op_rec(Id,
840 Dict),
841 
:-(
 Answer = case {Name, Op} of
842 {searchRequest, {searchResEntry, R}}
843 when is_record(R, 'SearchResultEntry') ->
844 
:-(
 New_dict = dict:append(Id, R, Dict),
845 
:-(
 {ok, S#eldap{dict = New_dict}};
846 {searchRequest, {searchResDone, Result}} ->
847 
:-(
 Reason = Result#'LDAPResult'.resultCode,
848 
:-(
 if Reason == success; Reason == sizeLimitExceeded ->
849 
:-(
 {Res, Ref} = polish(Result_so_far),
850 
:-(
 New_dict = dict:erase(Id, Dict),
851 
:-(
 misc:cancel_timer(Timer),
852 
:-(
 {reply,
853 #eldap_search_result{entries = Res,
854 referrals = Ref},
855 From, S#eldap{dict = New_dict}};
856 true ->
857 
:-(
 New_dict = dict:erase(Id, Dict),
858 
:-(
 misc:cancel_timer(Timer),
859 
:-(
 {reply, {error, Reason}, From,
860 S#eldap{dict = New_dict}}
861 end;
862 {searchRequest, {searchResRef, R}} ->
863 
:-(
 New_dict = dict:append(Id, R, Dict),
864 
:-(
 {ok, S#eldap{dict = New_dict}};
865 {addRequest, {addResponse, Result}} ->
866 
:-(
 New_dict = dict:erase(Id, Dict),
867 
:-(
 misc:cancel_timer(Timer),
868 
:-(
 Reply = check_reply(Result, From),
869 
:-(
 {reply, Reply, From, S#eldap{dict = New_dict}};
870 {delRequest, {delResponse, Result}} ->
871 
:-(
 New_dict = dict:erase(Id, Dict),
872 
:-(
 misc:cancel_timer(Timer),
873 
:-(
 Reply = check_reply(Result, From),
874 
:-(
 {reply, Reply, From, S#eldap{dict = New_dict}};
875 {modifyRequest, {modifyResponse, Result}} ->
876 
:-(
 New_dict = dict:erase(Id, Dict),
877 
:-(
 misc:cancel_timer(Timer),
878 
:-(
 Reply = check_reply(Result, From),
879 
:-(
 {reply, Reply, From, S#eldap{dict = New_dict}};
880 {modDNRequest, {modDNResponse, Result}} ->
881 
:-(
 New_dict = dict:erase(Id, Dict),
882 
:-(
 misc:cancel_timer(Timer),
883 
:-(
 Reply = check_reply(Result, From),
884 
:-(
 {reply, Reply, From, S#eldap{dict = New_dict}};
885 {bindRequest, {bindResponse, Result}} ->
886 
:-(
 New_dict = dict:erase(Id, Dict),
887 
:-(
 misc:cancel_timer(Timer),
888 
:-(
 Reply = check_bind_reply(Result, From),
889 
:-(
 {reply, Reply, From, S#eldap{dict = New_dict}};
890 {extendedReq, {extendedResp, Result}} ->
891 
:-(
 New_dict = dict:erase(Id, Dict),
892 
:-(
 misc:cancel_timer(Timer),
893 
:-(
 Reply = check_extended_reply(Result, From),
894 
:-(
 {reply, Reply, From, S#eldap{dict = New_dict}};
895 {OtherName, OtherResult} ->
896 
:-(
 New_dict = dict:erase(Id, Dict),
897 
:-(
 misc:cancel_timer(Timer),
898 
:-(
 {reply,
899 {error, {invalid_result, OtherName, OtherResult}},
900 From, S#eldap{dict = New_dict}}
901 end,
902 
:-(
 {response, Answer, Name};
903 
:-(
 Error -> Error
904 end.
